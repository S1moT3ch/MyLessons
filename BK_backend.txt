/**
 * Crea un calendario fisico sul tuo account, lo nomina per il docente e lo condivide.
 * Ritorna l'ID del calendario creato.
 */
function createAndShareTeacherCalendar(teacherEmail, teacherName) {
  try {
    const calName = "Agenda Lezioni - " + teacherName;
    
    // 1. Creazione
    const newCalendar = CalendarApp.createCalendar(calName, {
      summary: "Calendario sincronizzato con LessonApp per " + teacherName,
      timeZone: "Europe/Rome"
    });

    const calendarId = newCalendar.getId();
    console.log("Calendario creato: " + calendarId);

    // Aspettiamo 1 secondo per dare tempo a Google di registrare l'ID
    Utilities.sleep(1000);

    // 2. Condivisione ACL (Parametri corretti)
    if (teacherEmail) {
      const cleanEmail = teacherEmail.toLowerCase().trim();
      
      const resource = {
        role: 'writer', // <--- 'writer' è il termine corretto per i permessi di modifica nelle API
        scope: {
          type: 'user',
          value: cleanEmail
        }
      };

      // Chiamata API avanzata
      Calendar.Acl.insert(resource, calendarId, { sendNotifications: true });
      console.log("ACL inserita correttamente per: " + cleanEmail);
    }

    return calendarId;
  } catch (error) {
    console.error("Errore completo: " + error.toString());
    // Se fallisce l'ACL ma il calendario esiste, restituiamo comunque l'ID
    // così almeno viene scritto nella colonna H e puoi condividerlo a mano
    return (error.toString().includes("acl")) ? "ERROR_ACL_BUT_CREATED" : null;
  }
}

function syncSchedulesToCalendar(calendarId, teacherName, allSchedules) {
  if (!calendarId || !allSchedules) return;
  try {
    const calendar = CalendarApp.getCalendarById(calendarId);
    if (!calendar) return;

    // 1. Finestra temporale di sincronizzazione
    const now = new Date();
    const startWindow = new Date(now);
    const dayOfWeek = now.getDay(); 
    const diffToMonday = (dayOfWeek === 0) ? 0 : (dayOfWeek - 1);
    startWindow.setDate(now.getDate() - diffToMonday);
    startWindow.setHours(0, 0, 0, 0);
    const endWindow = new Date(startWindow);
    endWindow.setDate(startWindow.getDate() + 21); 

    // 2. Mappatura eventi esistenti
    const existingEvents = calendar.getEvents(startWindow, endWindow);
    const calendarMap = {};

    existingEvents.forEach(event => {
      const desc = event.getDescription() || "";
      if (desc.includes("[ID_LESSON_APP]")) {
        const match = desc.match(/\[FP:(.*?)\]/);
        if (match && match[1]) {
          calendarMap[match[1]] = event;
        }
      }
    });

    // 3. Ciclo sugli slot e creazione eventi INDIVIDUALI
    allSchedules.forEach(slot => {
      if (slot.email && slot.ora && slot.giorno) {
        const startTime = getStartDateFromSlot(slot.giorno, slot.ora);
        const endTime = new Date(startTime.getTime() + 60 * 60000);
        
        // Splittiamo le email per creare eventi separati
        const emails = slot.email.split(",").map(e => e.trim()).filter(e => e !== "");
        const nomi = (slot.nome || "").split(",").map(n => n.trim());

        emails.forEach((email, index) => {
          const studentName = nomi[index] || email;
          const cleanEmail = email.toLowerCase();
          const cleanTime = slot.ora.replace(/[^0-9]/g, '');
          const cleanDay = slot.giorno.toLowerCase().trim();
          
          // Il fingerprint ora è univoco per studente: include l'email del singolo
          const fingerprint = `FP_${teacherName}_${cleanDay}_${cleanTime}_${cleanEmail}`.replace(/[^a-zA-Z0-9_]/g, '');

          if (calendarMap[fingerprint]) {
            // Se l'evento esiste già per questo studente, lo manteniamo
            delete calendarMap[fingerprint];
          } else {
            // Creiamo l'evento individuale
            calendar.createEvent(`${studentName}`, startTime, endTime, {
              description: `Docente: ${teacherName}\nStudente: ${email}\n[ID_LESSON_APP]\n[FP:${fingerprint}]`,
              guests: email, // Un solo invitato per evento
              sendInvites: true
            }).addPopupReminder(0);
          }
        });
      }
    });

    // 4. CANCELLAZIONE: Rimuove gli eventi che non sono più nell'agenda (inclusi studenti rimossi da un gruppo)
    for (let fp in calendarMap) {
      console.log("RIMUOVO OBSOLETO: " + fp);
      calendarMap[fp].deleteEvent();
    }

  } catch (e) { 
    console.error("Errore Sync: " + e.toString()); 
  }
}

function getStartDateFromSlot(dayName, timeStr) {
  const days = ["Domenica", "Lunedì", "Martedì", "Mercoledì", "Giovedì", "Venerdì", "Sabato"];
  const targetIdx = days.indexOf(dayName);
  const now = new Date();
  const currentDayOfWeek = now.getDay(); // 0 è Domenica, 1 Lunedì...

  let baseDate = new Date(now);

  // LOGICA SETTIMANALE:
  // Se oggi è Domenica (0), la "settimana di riferimento" è già quella successiva.
  // Se oggi è da Lunedì (1) a Sabato (6), la "settimana di riferimento" è quella attuale.
  if (currentDayOfWeek === 0) {
    // Siamo di domenica: calcoliamo le date partendo dal Lunedì successivo
    // Portiamo la baseDate al Lunedì che sta per iniziare (domani)
    baseDate.setDate(now.getDate() + 1);
  } else {
    // Siamo tra Lunedì e Sabato: la baseDate deve essere il Lunedì appena passato
    baseDate.setDate(now.getDate() - (currentDayOfWeek - 1));
  }

  // Ora calcoliamo la data finale aggiungendo i giorni necessari rispetto al Lunedì di riferimento
  // targetIdx - 1 perché targetIdx 1 (Lunedì) deve dare offset 0
  const targetDate = new Date(baseDate);
  targetDate.setDate(baseDate.getDate() + (targetIdx - 1));

  // Impostazione orario
  const [h, m] = timeStr.split(':');
  targetDate.setHours(parseInt(h), parseInt(m), 0, 0);
  
  return targetDate;
}

function handleStudentFeedback(requestData) {
  try {
    const ssMain = getMainSS(); 
    let sheet = ssMain.getSheetByName("Feedback");
    
    // 1. Creazione/Controllo intestazioni (aggiungiamo la colonna E)
    if (!sheet) {
      sheet = ssMain.insertSheet("Feedback");
      sheet.appendRow(["ID_Univoco", "Status", "Nota", "Preferenza_Alternativa", "Data_Aggiornamento"]);
      sheet.getRange("1:1").setFontWeight("bold").setBackground("#f3f3f3");
    }

    const { studentEmail, teacherName, giorno, ora, status, note, preferenza } = requestData;
    
    const cleanEmail = studentEmail ? studentEmail.toLowerCase().trim() : "unknown";
    const cleanTime = ora ? ora.replace(/[^0-9]/g, '') : "0000";
    const uniqueKey = `${teacherName}-${giorno}-${cleanTime}-${cleanEmail}`;
    
    const data = sheet.getDataRange().getValues();
    let rowIndex = -1;

    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === uniqueKey) {
        rowIndex = i + 1;
        break;
      }
    }

    const timestamp = new Date();
    // La "preferenza" sarà una stringa tipo "Lunedì alle 15:00"
    let finalPref = "";
    if (preferenza) {
      // Se la preferenza contiene un'ora, ci assicuriamo che abbia il punto
      // Esempio: "Lunedì 1700" -> "Lunedì 17.00"
      finalPref = preferenza.replace(/(\d{2})(\d{2})/, "$1:$2");
    }

    if (rowIndex === -1) {
      // 2. Inserimento (Colonna A, B, C, D, E)
      sheet.appendRow([uniqueKey, status, note, finalPref, timestamp]);
    } else {
      // 3. Aggiornamento (Colonne B, C, D, E)
      // Range: riga trovata, colonna 2, 1 riga, 4 colonne totali
      sheet.getRange(rowIndex, 2, 1, 4).setValues([[status, note, finalPref, timestamp]]);
    }

    return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT);

  } catch (error) {
    return ContentService.createTextOutput("Error: " + error.toString()).setMimeType(ContentService.MimeType.TEXT);
  }
}

function handleGetTeacherFeedbackSummary(teacherName) {
  const ssMain = SpreadsheetApp.getActiveSpreadsheet(); // o la tua funzione getMainSS()
  const feedbackSheet = ssMain.getSheetByName("Feedback");
  const studentiSheet = ssMain.getSheetByName("Studenti");
  
  if (!feedbackSheet) return createJsonResponse({ status: "success", data: [] });
  
  // 1. Creiamo mappe per ID e per Email
  const idToNameMap = {};
  const emailToNameMap = {};
  
  if (studentiSheet) {
    const studentiData = studentiSheet.getDataRange().getValues();
    for (let i = 1; i < studentiData.length; i++) {
      const idGoogle = studentiData[i][0] ? studentiData[i][0].toString().trim() : ""; // Colonna A
      const email = studentiData[i][1] ? studentiData[i][1].toString().toLowerCase().trim() : ""; // Colonna B
      const nome = studentiData[i][2] ? studentiData[i][2].toString().trim() : "";      // Colonna C
      const cognome = studentiData[i][3] ? studentiData[i][3].toString().trim() : "";   // Colonna D
      
      const nomeCompleto = `${nome} ${cognome}`.trim();
      
      if (idGoogle) idToNameMap[idGoogle] = nomeCompleto;
      if (email) emailToNameMap[email] = nomeCompleto;
    }
  }

  const fbData = feedbackSheet.getDataRange().getValues();
  const teacherNameClean = teacherName.toLowerCase().trim();

  const result = [];
  for (let i = 1; i < fbData.length; i++) {
    const key = fbData[i][0].toString();
    
    if (key.toLowerCase().startsWith(teacherNameClean)) {
      const parts = key.split("-");
      const identifier = parts[parts.length - 1].trim().toLowerCase();
      
      const trovato = idToNameMap[identifier] || emailToNameMap[identifier];
      
      result.push({
        studentName: trovato || identifier,
        giorno: parts[1] || "",
        ora: parts[2] || "",
        status: fbData[i][1] || "",
        note: fbData[i][2] || "",
        preferenza: fbData[i][3] || "", // <--- LEGGI LA COLONNA D
        timestamp: fbData[i][4] || ""    // <--- IL TIMESTAMP SCALA ALLA COLONNA E (indice 4)
      });
    }
  }
  
  result.sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp));
  return createJsonResponse({ status: "success", data: result });
}

function verifyToken(idToken) {
  if (!idToken) return null;
  
  const cache = CacheService.getScriptCache();
  // Usiamo una versione "pulita" del token come chiave (max 250 caratteri per la cache)
  const cacheKey = "token_" + idToken.substring(idToken.length - 100).replace(/[^a-zA-Z0-9]/g, "");
  
  // 1. Controlla se il token è già stato validato di recente
  const cachedEmail = cache.get(cacheKey);
  if (cachedEmail) return cachedEmail;

  try {
    const response = UrlFetchApp.fetch("https://oauth2.googleapis.com/tokeninfo?id_token=" + idToken);
    const payload = JSON.parse(response.getContentText());
    
    if (payload.aud !== MY_CLIENT_ID) return null;
    
    const email = payload.email;
    // 2. Salva l'email in cache per 30 minuti (1800 secondi)
    cache.put(cacheKey, email, 1800);
    
    return email; 
  } catch (e) {
    console.error("Errore verifica token: " + e.toString());
    return null;
  }
}

function doGet(e) {
  const action = e.parameter.action;
  const idToken = e.parameter.token;
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

  try {
    // --- 1. AZIONE PUBBLICA: Verifica Codice Docente (Hash) ---
    if (action === "verifyTeacherCode") {
      const inputHash = e.parameter.code; 
      const savedHash = PropertiesService.getScriptProperties().getProperty('TEACHER_SECRET_CODE');
      return createJsonResponse({ 
        status: "success", 
        isValid: (inputHash && inputHash === savedHash) 
      });
    }

    // --- 2. AZIONE PUBBLICA: Elenco Insegnanti per Iscrizione ---
    if (action === "getTeachers") {
      const sheet = ss.getSheetByName("Insegnanti");
      const data = sheet.getDataRange().getValues();
      return createJsonResponse({ 
        status: "success", 
        data: data.slice(1).map(row => ({ id: row[0], email: row[1], name: row[2] + " " + row[3] }))
      });
    }

    // --- 3. VERIFICA LOGIN INIZIALE (checkUser) ---
    if (action === "checkUser") {
      const loginEmail = e.parameter.email;
      const sheets = ["Insegnanti", "Studenti"];
      let foundRole = null;
      for (let name of sheets) {
        let sheet = ss.getSheetByName(name);
        if (!sheet) continue;
        let data = sheet.getDataRange().getValues();
        let user = data.find(row => row[1] && row[1].toString().toLowerCase().trim() === loginEmail.toLowerCase().trim());
        if (user) { 
          foundRole = (name === "Insegnanti") ? "Insegnante" : "Studente"; 
          break; 
        }
      }
      return createJsonResponse({ exists: !!foundRole, role: foundRole });
    }

    // --- PROTEZIONE: Tutte le azioni seguenti richiedono Token Valido ---
    const authenticatedEmail = verifyToken(idToken);
    if (!authenticatedEmail) {
      return createJsonResponse({ status: "error", message: "Non autorizzato: Sessione non valida" });
    }

    if (action === "getStudentPersonalSchedule") {
      const studentEmail = authenticatedEmail.toLowerCase().trim();
      
      // 1. Recuperiamo tutti i feedback dello studente dal foglio "Feedback"
      const ssMain = getMainSS();
      const feedbackSheet = ssMain.getSheetByName("Feedback");
      const feedbackMap = {};
      
      if (feedbackSheet) {
        const fbData = feedbackSheet.getDataRange().getValues();
        // La chiave nel foglio Feedback è: Docente-Giorno-Ora-Email
        for (let i = 1; i < fbData.length; i++) {
          const key = fbData[i][0];
          if (key.includes(studentEmail)) {
            feedbackMap[key] = {
              status: fbData[i][1],
              note: fbData[i][2],
              preferenza: fbData[i][3]
            };
          }
        }
      }

      // 2. Recuperiamo gli slot orari (logica che avevi già)
      const ssSchedule = getScheduleSS();
      const sheets = ssSchedule.getSheets();
      const result = [];

      sheets.forEach(sheet => {
        const data = sheet.getDataRange().getDisplayValues();
        if (data.length < 2) return;
        const teacherName = sheet.getName();

        Object.keys(DAY_MAP).forEach(giorno => {
          const colOra = DAY_MAP[giorno];
          const colStudente = colOra + 1;

          for (let r = 1; r < data.length; r++) {
            const cellaEmailRaw = data[r][colStudente] ? data[r][colStudente].toLowerCase() : "";
            const listaEmail = cellaEmailRaw.split(",").map(e => e.trim());

            if (listaEmail.includes(studentEmail)) {
              const ora = data[r][colOra].toString().trim();
              const cleanTime = ora.replace(/[^0-9]/g, '');
              
              // GENERIAMO LA CHIAVE PER IL CONFRONTO
              const uniqueKey = `${teacherName}-${giorno}-${cleanTime}-${studentEmail}`;
              
              // UNIAMO IL FEEDBACK SE ESISTE, ALTRIMENTI "In attesa"
              const fb = feedbackMap[uniqueKey] || { status: "In attesa", note: "" };

              result.push({ 
                giorno: giorno, 
                ora: ora, 
                teacherName: teacherName,
                feedbacks: [fb] // Lo passiamo come array per coerenza con il frontend
              });
            }
          }
        });
      });

      // Ordinamento per giorno
      const dayOrder = ["Lunedì", "Martedì", "Mercoledì", "Giovedì", "Venerdì", "Sabato"];
      result.sort((a, b) => dayOrder.indexOf(a.giorno) - dayOrder.indexOf(b.giorno));

      return createJsonResponse({ status: "success", data: result });
    }

    // --- 5. ISCRIZIONI DELLO STUDENTE (Quali corsi segue) ---
    if (action === "getMySubscriptions") {
      const sheet = ss.getSheetByName("Iscrizioni");
      if (!sheet) return createJsonResponse({ status: "success", data: [] });
      const data = sheet.getDataRange().getValues();
      const mySubs = data.slice(1)
        .filter(row => row[1].toString().toLowerCase().trim() === authenticatedEmail.toLowerCase().trim())
        .map(row => ({ 
          teacherId: row[2], 
          teacherName: row[3], 
          date: row[4] 
        }));
      return createJsonResponse({ status: "success", data: mySubs });
    }

    // --- 6. STUDENTI ISCRITTI (Vista Docente - Ottimizzata) ---
    if (action === "getTeacherSubscribers") {
      const teacherId = e.parameter.teacherId;
      const sheetIscrizioni = ss.getSheetByName("Iscrizioni");
      const sheetStudenti = ss.getSheetByName("Studenti");
      
      if (!sheetIscrizioni || !sheetStudenti) {
        return createJsonResponse({ status: "success", data: [] });
      }

      // 1. Lettura BATCH dei dati anagrafici (una sola chiamata al server)
      const iscrizioniData = sheetIscrizioni.getDataRange().getValues();
      const studentiData = sheetStudenti.getDataRange().getValues().slice(1);
      
      // Creazione Mappa Nomi in memoria: velocizza il lookup degli studenti

      studentiData.forEach(r => {

      // 2. Conteggio settimanale efficiente (Scansione file Orari)

      const allScheduleSheets = ssSchedule.getSheets();
      const weeklyCounts = {};

      // Indici delle colonne email basati sul tuo DAY_MAP (es. 1, 3, 5, 7, 9, 11)
      const emailColumnIndices = Object.values(DAY_MAP).map(colIdx => colIdx + 1);

      allScheduleSheets.forEach(s => {
        const data = s.getDataRange().getValues();
        if (data.length < 2) return; // Salta fogli vuoti o solo intestazione

        // Processamento in RAM: evitiamo cicli for pesanti sulle celle
        for (let r = 1; r < data.length; r++) {
          emailColumnIndices.forEach(c => {
            const email = data[r][c] ? data[r][c].toString().toLowerCase().trim() : "";
            if (email && email.includes("@")) {
              weeklyCounts[email] = (weeklyCounts[email] || 0) + 1;
            }
          });
        }
      });

      // 3. Filtraggio e costruzione del JSON finale
      const result = iscrizioniData.slice(1)
        .filter(row => row[2].toString() === teacherId.toString())
        .map(row => {
          const email = row[1].toLowerCase().trim();
          return {
            studentName: nomiStudenti[email] || "Sconosciuto",
            studentEmail: row[1],
            date: row[4],
            lessonCount: weeklyCounts[email] || 0, // Ore totali assegnate questa settimana
            lezioniSvolte: Number(row[5]) || 0,     // Storico colonna F
            lezioniDaPagare: Number(row[6]) || 0,  // Storico colonna G
            tariffa: Number(row[7]) || 0           // Colonna H
          };
        });

      return createJsonResponse({ status: "success", data: result });
    }

    // --- 7. AGENDA COMPLETA DOCENTE (Ottimizzata per studenti illimitati) ---
    if (action === "getStudentSchedules") {
      const teacherName = e.parameter.teacherName;
      const ssSchedule = SpreadsheetApp.openById(SCHEDULE_FILE_ID);
      const sheetOrari = getTeacherSheet(ssSchedule, teacherName);
      
      // 1. Caricamento BATCH dei dati dell'orario
      const data = sheetOrari.getDataRange().getValues();
      if (data.length < 2) return createJsonResponse({ status: "success", data: [] });

      // 2. Recupero Nomi Studenti (Ottimizzato con Cache)
      const cache = CacheService.getScriptCache();
      let nomiMap = JSON.parse(cache.get("student_names_map") || "null");

      if (!nomiMap) {
        const ssMain = SpreadsheetApp.openById(SPREADSHEET_ID);
        const studentiSheet = ssMain.getSheetByName("Studenti");
        const studentiData = studentiSheet ? studentiSheet.getDataRange().getValues().slice(1) : [];
        nomiMap = {};
        studentiData.forEach(r => { 
          if(r[1]) nomiMap[r[1].toLowerCase().trim()] = r[2] + " " + r[3]; 
        });
        cache.put("student_names_map", JSON.stringify(nomiMap), 3600);
      }

      const result = [];
      const days = Object.keys(DAY_MAP);

      // 3. Processamento efficiente in RAM con gestione MULTI-STUDENTE
      days.forEach(giorno => {
        const colOra = DAY_MAP[giorno];
        const colStudente = colOra + 1;

        for (let r = 1; r < data.length; r++) {
          const oraVal = data[r][colOra];
          if (oraVal) {
            const emailsStr = data[r][colStudente] ? data[r][colStudente].toString() : "";
            
            // --- LOGICA MULTI-STUDENTE ---

            const emailList = emailsStr.split(",")
              .map(e => e.trim())
              .filter(e => e !== "");

            // Creiamo l'array di oggetti richiesto dal nuovo frontend ottimizzato
            const studentsArray = emailList.map(email => ({
              email: email,
              nome: nomiMap[email.toLowerCase()] || email
            }));

            result.push({
              giorno: giorno,
              ora: oraVal.toString().trim(),
              email: emailsStr, // Manteniamo la stringa per compatibilità con vecchi componenti
              students: studentsArray, // Array per i Chip dinamici del frontend
              nome: studentsArray.length > 0 ? studentsArray[0].nome : "" // Fallback primo nome
            });
          }
        }
      });

      return createJsonResponse({ status: "success", data: result });
    }

    if (action === "getStudentBalances") {
      const studentEmail = e.parameter.studentEmail.toLowerCase().trim();
      const sheetIscrizioni = ssMain.getSheetByName("Iscrizioni");
      const data = sheetIscrizioni.getDataRange().getValues().slice(1);
      
      // Mappa degli insegnanti per convertire l'ID nel Nome reale (se presente)
      const sheetInsegnanti = ssMain.getSheetByName("Insegnanti");
      const insegnantiData = sheetInsegnanti ? sheetInsegnanti.getDataRange().getValues() : [];
      const nomiInsegnanti = {};
      insegnantiData.forEach(r => nomiInsegnanti[r[0]] = r[1]); 

      const result = data
        .filter(row => row[1].toString().toLowerCase().trim() === studentEmail)
        .map(row => {
          const lezioniDaPagare = Number(row[6]) || 0; // Col G
          const tariffa = Number(row[7]) || 0;         // Col H
          
          return {
            teacherName: row[3] || nomiInsegnanti[row[2]] || "Insegnante", // Col D o lookup
            lezioniSvolte: Number(row[5]) || 0,        // Col F
            lezioniDaPagare: lezioniDaPagare,          // Col G
            amountDue: lezioniDaPagare * tariffa       // <-- CALCOLO ESEGUITO QUI
          };
        });
        
      return createJsonResponse({ status: "success", data: result });
    }

    if (action === "getTeacherFeedbackSummary") {
      return handleGetTeacherFeedbackSummary(e.parameter.teacherName)
    }

    return createJsonResponse({ status: "error", message: "Azione non riconosciuta" });

  } catch (error) { 
    return createJsonResponse({ status: "error", message: error.toString() }); 
  }
  
}

function doPost(e) {
  console.log("--- NUOVA RICHIESTA POST RICEVUTA ---");
  try {
    const requestData = JSON.parse(e.postData.contents);
    const ssMain = SpreadsheetApp.openById(SPREADSHEET_ID);
    const action = requestData.action;
    const authenticatedEmail = verifyToken(requestData.id_token);

    if (action === "saveFullSchedule") {
      if (!authenticatedEmail) return createTextResponse("Error: Non autorizzato");
      
      const ssSchedule = getScheduleSS();
      const teacherName = requestData.teacherName;
      const sheet = getTeacherSheet(ssSchedule, teacherName);
      const ssMain = getMainSS();
      const sheetIscrizioni = ssMain.getSheetByName("Iscrizioni");
      const allSchedules = requestData.allSchedules; // Array di oggetti con {giorno, ora, email (che contiene stringa con virgole)}

      // 1. Lettura BATCH iniziale
      const oldData = sheet.getDataRange().getValues();
      const iscrizioniRange = sheetIscrizioni.getDataRange();
      const iscrizioniValues = iscrizioniRange.getValues();
      const insData = ssMain.getSheetByName("Insegnanti").getDataRange().getValues();
      
      const teacherRow = insData.find(r => r[1] === authenticatedEmail);
      const calendarId = teacherRow ? teacherRow[7] : null;

      // 2. Mappa situazione attuale (Splittiamo le email esistenti per il confronto)
      const oldAssignments = {}; 
      for (let giorno in DAY_MAP) {
        let colOra = DAY_MAP[giorno];
        for (let r = 1; r < oldData.length; r++) {
          let ora = oldData[r][colOra];
          let emailsStr = oldData[r][colOra + 1] ? oldData[r][colOra + 1].toString().toLowerCase().trim() : "";
          if (ora && emailsStr) {
            // Creiamo un array di email per quello slot
            oldAssignments[`${giorno}-${ora}`] = emailsStr.split(",").map(e => e.trim());
          }
        }
      }

      // --- 3. RICALCOLO TOTALE CONTATORI (Invece di netChanges) ---
      // Invece di calcolare la differenza, contiamo quante volte ogni email appare nel NUOVO orario
      const newCounts = {};
      allSchedules.forEach(item => {
          if (item.email) {
              const emails = item.email.split(",").map(e => e.trim().toLowerCase());
              emails.forEach(email => {
                  if (email) newCounts[email] = (newCounts[email] || 0) + 1;
              });
          }
      });

      // Contiamo quante volte ogni email appariva nel VECCHIO orario
      const oldCounts = {};
      for (let key in oldAssignments) {
          oldAssignments[key].forEach(email => {
              if (email) oldCounts[email] = (oldCounts[email] || 0) + 1;
          });
      }

      // Calcoliamo la differenza reale tra nuovo e vecchio conteggio per ogni studente
      const studentEmails = new Set([...Object.keys(newCounts), ...Object.keys(oldCounts)]);
      const finalDiffs = {};
      studentEmails.forEach(email => {
          const diff = (newCounts[email] || 0) - (oldCounts[email] || 0);
          if (diff !== 0) finalDiffs[email] = diff;
      });

      // 4. Aggiornamento BATCH foglio Iscrizioni usando finalDiffs
      if (Object.keys(finalDiffs).length > 0) {
          for (let i = 1; i < iscrizioniValues.length; i++) {
              const email = iscrizioniValues[i][1].toString().toLowerCase().trim();
              if (finalDiffs[email]) {
                  // Aggiorniamo basandoci sulla differenza di conteggio totale
                  iscrizioniValues[i][5] = Math.max(0, (Number(iscrizioniValues[i][5]) || 0) + finalDiffs[email]); 
                  iscrizioniValues[i][6] = Math.max(0, (Number(iscrizioniValues[i][6]) || 0) + finalDiffs[email]);
              }
          }
          iscrizioniRange.setValues(iscrizioniValues);
      }

      // 5. Preparazione BATCH foglio Orari (12 colonne standard)
      const newScheduleMatrix = Array.from({ length: 50 }, () => Array(12).fill(""));
      
      allSchedules.forEach(item => {
        const startCol = DAY_MAP[item.giorno];
        for (let r = 0; r < 50; r++) {
          if (!newScheduleMatrix[r][startCol]) {
            newScheduleMatrix[r][startCol] = item.ora;
            // Salviamo la stringa con virgole (es: "mario@gmail.com, luca@gmail.com")
            newScheduleMatrix[r][startCol + 1] = item.email; 
            break;
          }
        }
      });
      sheet.getRange(2, 1, 50, 12).setValues(newScheduleMatrix);

      // 6. Sincronizzazione Calendario
      if (calendarId) {
        // Nota: allSchedules deve contenere gli studenti splittati per creare eventi singoli
        syncSchedulesToCalendar(calendarId, teacherName, allSchedules);
      }

      PropertiesService.getScriptProperties().setProperty('LAST_SCHEDULE_UPDATE', Array.from(new Date().getTime().toString()));
      PropertiesService.getScriptProperties().setProperty('last_update_all', new Date().getTime().toString());

      return createTextResponse("Success");
    }

    if (action === "updatePaidLessons") {
        if (!authenticatedEmail) return createTextResponse("Error: Non autorizzato");
        const sheet = ssMain.getSheetByName("Iscrizioni");
        const data = sheet.getDataRange().getValues();
        let rowIndex = -1;
        for (let i = 1; i < data.length; i++) {
          if (data[i][1].toString().toLowerCase().trim() === requestData.studentEmail.toLowerCase().trim() && data[i][2].toString() === requestData.teacherId.toString()) {
            rowIndex = i + 1; break;
          }
        }
        if (rowIndex !== -1) {
          sheet.getRange(rowIndex, 7).setValue(requestData.newPaidValue);
          return createTextResponse("Success");
        }
        return createTextResponse("Error: Non trovato");
      }

        if (action === "updateStudentRate") {
          if (!authenticatedEmail) return createTextResponse("Error: Non autorizzato");
          const sheet = ssMain.getSheetByName("Iscrizioni");
          const data = sheet.getDataRange().getValues();
          let rowIndex = -1;

          for (let i = 1; i < data.length; i++) {
              if (data[i][1].toString().toLowerCase().trim() === requestData.studentEmail.toLowerCase().trim() && 
                  data[i][2].toString() === requestData.teacherId.toString()) {
                  rowIndex = i + 1; 
                  break;
              }
          }

          if (rowIndex !== -1) {
              // Colonna H è l'ottava colonna (indice 8)
              sheet.getRange(rowIndex, 8).setValue(requestData.newRate);
              return createTextResponse("Success");
          }
          return createTextResponse("Error: Non trovato");
    }

    if (action === "resetScheduleForNewWeek") {
      if (!authenticatedEmail) return createTextResponse("Error: Non autorizzato");
      
      const ssSchedule = SpreadsheetApp.openById(SCHEDULE_FILE_ID);
      const teacherName = requestData.teacherName;
      const sheet = getTeacherSheet(ssSchedule, teacherName);
      
      // 1. Pulizia fisica del foglio (colonne email)
      for (let giorno in DAY_MAP) {
        let colEmailIndex = DAY_MAP[giorno] + 2; 
        sheet.getRange(2, colEmailIndex, 50, 1).clearContent();
      }

      // 2. Recupero del CalendarID per sincronizzare la cancellazione
      const ssMain = SpreadsheetApp.openById(SPREADSHEET_ID);
      const sheetInsegnanti = ssMain.getSheetByName("Insegnanti");
      const insData = sheetInsegnanti.getDataRange().getValues();
      let calendarId = null;
      
      for(let i=1; i<insData.length; i++) {
        if (insData[i][1] === authenticatedEmail) {
          calendarId = insData[i][7]; // Colonna H
          break;
        }
      }

      // 3. Sincronizzazione col Calendario
      if (calendarId) {
        // Passiamo un array vuoto [] come allSchedules. 
        // La funzione syncSchedulesToCalendar vedrà che non ci sono più slot nel "nuovo" schedule
        // e procederà a eliminare tutti gli eventi con [ID_LESSON_APP] dal calendario.
        syncSchedulesToCalendar(calendarId, teacherName, []);
        console.log("Calendario resettato per: " + teacherName);
      }

      PropertiesService.getScriptProperties().setProperty('last_update_all', new Date().getTime().toString());
      return createTextResponse("Success");
    }

    if (action === "removeSlotAndDecrement") {
      const data = JSON.parse(e.postData.contents);
      const { studentEmail, teacherName } = data;
      
      // CREA UNA CHIAVE DI IDEMPOTENZA
      const lockKey = "lock_" + studentEmail + "_" + teacherName;
      const scriptProperties = PropertiesService.getScriptProperties();
      const lastCall = scriptProperties.getProperty(lockKey);
      const now = new Date().getTime();

      // Se l'ultima chiamata è avvenuta meno di 2 secondi fa, ignoriamo questa
      if (lastCall && (now - parseInt(lastCall)) < 2000) {
        return createTextResponse("Ignored: Duplicated request");
      }
      scriptProperties.setProperty(lockKey, now.toString());
      
      if (!studentEmail) return createTextResponse("Success: Slot vuoto rimosso");

      const ss = SpreadsheetApp.getActiveSpreadsheet();
      const sheetIscrizioni = ss.getSheetByName("Iscrizioni");
      const rows = sheetIscrizioni.getDataRange().getValues();
      
      for (let i = 1; i < rows.length; i++) {
        // Cerchiamo la riga che corrisponde a Studente + Insegnante
        if (rows[i][1] === studentEmail && rows[i][3] === teacherName) {
          // Decrementiamo Svolte (Col F/5) e Da Pagare (Col G/6)
          let svolte = Math.max(0, rows[i][5] - 1);
          let daPagare = Math.max(0, rows[i][6] - 1);
          
          sheetIscrizioni.getRange(i + 1, 6).setValue(svolte);
          sheetIscrizioni.getRange(i + 1, 7).setValue(daPagare);
          break;
        }
      }
      PropertiesService.getScriptProperties().setProperty('last_update_all', new Date().getTime().toString());
      return createTextResponse("Success: Contatori aggiornati");
    }

    if (action === "updateStudentFeedback") {
      return handleStudentFeedback(requestData);
    }

    if (action === "subscribe") {
      if (!authenticatedEmail) return createTextResponse("Error: Non autorizzato");
      CacheService.getScriptCache().remove("student_names_map");
      const sheet = ssMain.getSheetByName("Iscrizioni");
      sheet.appendRow([requestData.studentId, authenticatedEmail, requestData.teacherId, requestData.teacherName, new Date(), 0, 0]);
      return createTextResponse("Success");
    }

    if (requestData.id_token && !action) {
      console.log("Entrato nel blocco Registrazione/Login");
      const tokenResponse = UrlFetchApp.fetch("https://oauth2.googleapis.com/tokeninfo?id_token=" + requestData.id_token);
      const googleUser = JSON.parse(tokenResponse.getContentText());
      
      const role = requestData.role;
      const teacherEmail = googleUser.email;
      const teacherFullName = googleUser.given_name + " " + googleUser.family_name;

      let sheet = ssMain.getSheetByName(role === "Insegnante" ? "Insegnanti" : "Studenti");
      const data = sheet.getDataRange().getValues();
      let userRowIndex = -1;

      // 1. Cerchiamo se l'utente esiste già
      for (let i = 0; i < data.length; i++) { 
        if (data[i][0] === googleUser.sub) { 
          userRowIndex = i + 1; 
          break; 
        } 
      }

      // 2. AZIONE SOLO SE L'UTENTE È NUOVO (REGISTRAZIONE)
      if (userRowIndex === -1) {
        console.log("Nuovo utente! Scrivo i dati per la prima e unica volta.");

        let calId = "";
        if (role === "Insegnante") {
          // Creazione calendario solo alla prima registrazione
          calId = createAndShareTeacherCalendar(teacherEmail, teacherFullName);
          
          // Inizializzazione foglio orari fisico
          const ssSchedule = SpreadsheetApp.openById(SCHEDULE_FILE_ID);
          getTeacherSheet(ssSchedule, teacherFullName);
        }

        // Prepariamo la riga completa
        const rowData = [
          googleUser.sub,          // Col A
          teacherEmail,            // Col B
          googleUser.given_name,   // Col C
          googleUser.family_name,  // Col D
          googleUser.picture,      // Col E
          new Date(),              // Col F (Data Iscrizione fissa)
          role,                    // Col G
          calId || ""              // Col H
        ];

        // Scrittura fisica sul foglio: avviene SOLO ORA
        sheet.appendRow(rowData);
        
      } else {
        // 3. LOGICA DI LOGIN SUCCESSIVO
        // Non facciamo nulla: il foglio non viene aperto in scrittura.
        // L'utente esiste già, quindi restituiamo semplicemente Success al frontend.
        console.log("Utente già presente. Dati anagrafici non modificati.");
      }

      return createTextResponse("Success");
    }
  } catch (error) { return createTextResponse("Error: " + error.toString()); }
}

function createTextResponse(msg) { return ContentService.createTextOutput(msg).setMimeType(ContentService.MimeType.TEXT); }
function createJsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
